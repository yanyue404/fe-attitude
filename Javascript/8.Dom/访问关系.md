/**
 * 功能:根据索引值找兄弟节点
 * @param ele
 * @param index
 * @returns {*}
 */
function getSibEleOfIndex(ele, index) {
    return ele.parentNode.children[index];
}

//拓展方法2
/**
 * 功能:查找所有兄弟节点(不包括自己)
 * @param ele
 * @returns {Array}
 */
function siblings(ele) {
    var newArr = [];
    var arr = ele.parentNode.children;

    for (var i = 0; i < arr.length; i++) {
        if (arr[i].nodeType == 1 && arr[i] != ele) {
            newArr.push(arr[i]);
        }
    }
    return newArr;
}
